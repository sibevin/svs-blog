pmap

defmodule Pmap do
  def pmap(list, f) do
    parent = self()
    pids = Enum.map(list, fn(el) -> spwan(fn -> send(parent, f.(el))end)end)
    gather(pids)
  end

  def gather([]), do: []
  def gather([head|tails]) do
    recive do
      {head, return} -> [return | gather(tails)]
  end
end
